 # CPU.js

## Syntax

* 0x00 = "add"
	* 0x01 = "and"
	* 0x02 = "call"
	* 0x03 = "cmp"
	* 0x04 = "cnst"
	* 0x05 = "ctrl"
	* 0x06 = "div"
	* 0x07 = "epm"
	* 0x08 = "hlt"
	* 0x09 = "insp"
	* 0x0A = int
	* 0x0B = jmpeq
	* 0x0C = jmpge
	* 0x0D = jmpneq
	* 0x0E = jmpsml
	* 0x0F = load
	* 0x10 = mov
	* 0x11 = mul
	* 0x12 = nop
	* 0x13 = not
	* 0x14 = or
	* 0x15 = pload
	* 0x16 = pop
	* 0x17 = pstore
	* 0x18 = push
	* 0x19 = ret
	* 0x1A = store
	* 0x1B = sub
	* 0x1C = xor

## Usage:
nodejs unit <PATH/TO/EXECUTABLE>
